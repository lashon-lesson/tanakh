גירוש הגר וישמעאל וברית אבימלך

בראשית כ״א — לידת יצחק, "מְצַחֵק", גירוש הגר וישמעאל, ההצלה במדבר, וברית באר שבע

⏱ 45 דק׳📝 תרגול + בגרות📜 סיפורי האבות
הערה למורה · רגישות

פרק זה משלים את סיפור הגר וישמעאל שנפתח בשיעור 4 (פרק ט״ז) — מומלץ לחבר במפורש בין השיעורים. שני נושאים דורשים תשומת לב: (1) המילה "מְצַחֵק" (פס׳ ט) פתוחה למגוון פירושים, מהקל (ליגלוג) עד החמור (רש״י); מומלץ להציג את הטווח הפרשני מבלי לקבוע עמדה חד-משמעית על אופיו של ישמעאל. (2) סצנת הייאוש של הגר במדבר (פס׳ טו–טז) רגשית וקשה — יש לגשת אליה בכובד ראש ורגישות, תוך הדגשת ההצלה שבאה מיד לאחריה.

👶
1
לידת יצחק ומימוש ההבטחה

רובד הפשט
"וַה׳ פָּקַד אֶת שָׂרָה כַּאֲשֶׁר אָמָר" — לידת יצחק היא מימוש מדויק של הבטחות הפרקים הקודמים (ט״ו, י״ז, י״ח). אברהם מל את יצחק בן שמונה ימים, כמצוות ברית המילה. שרה מכריזה: "צְחֹק עָשָׂה לִי אֱלֹהִים" — שמחה שהופכת לשם הבן.

⚖️
2
"מְצַחֵק" ודרישת הגירוש

במשתה הגמילה, "וַתֵּרֶא שָׂרָה אֶת בֶּן הָגָר הַמִּצְרִית... מְצַחֵק". שרה דורשת: "גָּרֵשׁ הָאָמָה הַזֹּאת וְאֶת בְּנָהּ כִּי לֹא יִירַשׁ בֶּן הָאָמָה הַזֹּאת עִם בְּנִי עִם יִצְחָק".

רובד הפרשנות · העשרה
מגוון פירושים ל"מְצַחֵק": רש״י מציג קריאה חמורה (עבודה זרה, ליצנות, או יריבות אלימה על הירושה); קריאות מתונות יותר רואות בכך לגלוג/זלזול ילדותי במעמדו הנחות של יצחק. הטקסט עצמו אינו מפרש — עמימות מכוונת.
💔
3
תגובת אברהם ותשובת ה׳

"וַיֵּרַע הַדָּבָר מְאֹד בְּעֵינֵי אַבְרָהָם עַל אוֹדֹת בְּנוֹ" — הכתוב מכנה את ישמעאל "בְּנוֹ", בניגוד ל"בֶּן הָאָמָה" בפי שרה. ה׳ מכריע כדעת שרה, אך מבטיח: "וְגַם אֶת בֶּן הָאָמָה לְגוֹי אֲשִׂימֶנּוּ כִּי זַרְעֲךָ הוּא" — שתי הבטחות מקבילות, ליצחק ולישמעאל.

🏜️
4
המשלוח, הייאוש וההצלה

אברהם משכים ונותן להגר לחם וחמת מים; היא תועה במדבר באר שבע. כשכלים המים, הגר משליכה את הילד תחת שיח ויושבת מנגד, "כִּי אָמְרָה אַל אֶרְאֶה בְּמוֹת הַיָּלֶד". ואז — "וַיִּשְׁמַע אֱלֹהִים אֶת קוֹל הַנַּעַר", מלאך קורא לה, ופוקח את עיניה לראות באר מים.

הד לפרק ט״ז: השם "ישמעאל" ("כי שמע ה׳") מתממש כאן במפורש, ומוטיב הראייה חוזר ("וַיִּפְקַח אֱלֹהִים אֶת עֵינֶיהָ") — כמו "אֵל רֳאִי" בפרק הקודם על הגר.
🤝
5
ברית אבימלך ובאר שבע

אבימלך פונה לאברהם: "אֱלֹהִים עִמְּךָ בְּכֹל אֲשֶׁר אַתָּה עֹשֶׂה" ומבקש ברית שלום. אברהם מעמיד שבע כבשות כעדות לחפירת הבאר; המקום נקרא בְּאֵר שֶׁבַע, ואברהם נוטע שם אשל וקורא בשם ה׳.

בארות במזרח הקדום: באזור מדברי-למחצה כארץ כנען, באר מים הייתה נכס כלכלי ואסטרטגי יקר-ערך — מקור מים קבוע קבע למעשה אילו עדרים ואילו שבטים יכולים להתקיים באזור. סכסוכי בארות (כמו זה שבין אברהם לעבדי אבימלך, פס׳ כה) היו לכן עניין למשא ומתן מדיני של ממש, לא מטרד שולי — וההסכם המשפטי הפומבי סביב הבאר מקביל במהותו למשא ומתן על מערת המכפלה (פרק כ״ג). לא במקרה יחזור סכסוך בארות דומה גם בימי יצחק, סביב אותו אזור עצמו (פרק כ״ו).
🧭 חמישה צירים לזכירת הפרק
1
לידה: יצחק נולד, נימול — מימוש ההבטחה.
2
"מצחק": שרה רואה, דורשת גירוש מחשש ירושה.
3
שתי הבטחות: "ביצחק יקרא לך זרע" + "בן האמה לגוי".
4
הצלה: ייאוש במדבר ← "שמע אלהים... ויפקח" ← באר מים.
5
ברית: אבימלך ← שבע כבשות ← באר שבע.

בראשית כ״א — חלק א׳: לידת יצחק והגירוש (א–יד)

א וַה׳ פָּקַד אֶת שָׂרָה כַּאֲשֶׁר אָמָר וַיַּעַשׂ ה׳ לְשָׂרָה כַּאֲשֶׁר דִּבֵּר. ב וַתַּהַר וַתֵּלֶד שָׂרָה לְאַבְרָהָם בֵּן לִזְקֻנָיו לַמּוֹעֵד אֲשֶׁר דִּבֶּר אֹתוֹ אֱלֹהִים. ג וַיִּקְרָא אַבְרָהָם אֶת שֶׁם בְּנוֹ הַנּוֹלַד לוֹ אֲשֶׁר יָלְדָה לּוֹ שָׂרָה יִצְחָק. ד וַיָּמָל אַבְרָהָם אֶת יִצְחָק בְּנוֹ בֶּן שְׁמֹנַת יָמִים כַּאֲשֶׁר צִוָּה אֹתוֹ אֱלֹהִים. ו וַתֹּאמֶר שָׂרָה צְחֹק עָשָׂה לִי אֱלֹהִים כָּל הַשֹּׁמֵעַ יִצְחַק לִי. ז וַתֹּאמֶר מִי מִלֵּל לְאַבְרָהָם הֵינִיקָה בָנִים שָׂרָה כִּי יָלַדְתִּי בֵן לִזְקֻנָיו. ח וַיִּגְדַּל הַיֶּלֶד וַיִּגָּמַל וַיַּעַשׂ אַבְרָהָם מִשְׁתֶּה גָדוֹל בְּיוֹם הִגָּמֵל אֶת יִצְחָק. ט וַתֵּרֶא שָׂרָה אֶת בֶּן הָגָר הַמִּצְרִית אֲשֶׁר יָלְדָה לְאַבְרָהָם מְצַחֵק. י וַתֹּאמֶר לְאַבְרָהָם גָּרֵשׁ הָאָמָה הַזֹּאת וְאֶת בְּנָהּ כִּי לֹא יִירַשׁ בֶּן הָאָמָה הַזֹּאת עִם בְּנִי עִם יִצְחָק. יא וַיֵּרַע הַדָּבָר מְאֹד בְּעֵינֵי אַבְרָהָם עַל אוֹדֹת בְּנוֹ. יב וַיֹּאמֶר אֱלֹהִים אֶל אַבְרָהָם אַל יֵרַע בְּעֵינֶיךָ עַל הַנַּעַר וְעַל אֲמָתֶךָ כֹּל אֲשֶׁר תֹּאמַר אֵלֶיךָ שָׂרָה שְׁמַע בְּקֹלָהּ כִּי בְיִצְחָק יִקָּרֵא לְךָ זָרַע. יג וְגַם אֶת בֶּן הָאָמָה לְגוֹי אֲשִׂימֶנּוּ כִּי זַרְעֲךָ הוּא. יד וַיַּשְׁכֵּם אַבְרָהָם בַּבֹּקֶר וַיִּקַּח לֶחֶם וְחֵמַת מַיִם וַיִּתֵּן אֶל הָגָר שָׂם עַל שִׁכְמָהּ וְאֶת הַיֶּלֶד וַיְשַׁלְּחֶהָ וַתֵּלֶךְ וַתֵּתַע בְּמִדְבַּר בְּאֵר שָׁבַע.

בראשית כ״א — חלק ב׳: הייאוש וההצלה במדבר (טו–כא)

טו וַיִּכְלוּ הַמַּיִם מִן הַחֵמֶת וַתַּשְׁלֵךְ אֶת הַיֶּלֶד תַּחַת אַחַד הַשִּׂיחִם. טז וַתֵּלֶךְ וַתֵּשֶׁב לָהּ מִנֶּגֶד הַרְחֵק כִּמְטַחֲוֵי קֶשֶׁת כִּי אָמְרָה אַל אֶרְאֶה בְּמוֹת הַיָּלֶד וַתֵּשֶׁב מִנֶּגֶד וַתִּשָּׂא אֶת קֹלָהּ וַתֵּבְךְּ. יז וַיִּשְׁמַע אֱלֹהִים אֶת קוֹל הַנַּעַר וַיִּקְרָא מַלְאַךְ אֱלֹהִים אֶל הָגָר מִן הַשָּׁמַיִם וַיֹּאמֶר לָהּ מַה לָּךְ הָגָר אַל תִּירְאִי כִּי שָׁמַע אֱלֹהִים אֶל קוֹל הַנַּעַר בַּאֲשֶׁר הוּא שָׁם. יח קוּמִי שְׂאִי אֶת הַנַּעַר וְהַחֲזִיקִי אֶת יָדֵךְ בּוֹ כִּי לְגוֹי גָּדוֹל אֲשִׂימֶנּוּ. יט וַיִּפְקַח אֱלֹהִים אֶת עֵינֶיהָ וַתֵּרֶא בְּאֵר מָיִם וַתֵּלֶךְ וַתְּמַלֵּא אֶת הַחֵמֶת מַיִם וַתַּשְׁקְ אֶת הַנָּעַר. כ וַיְהִי אֱלֹהִים אֶת הַנַּעַר וַיִּגְדָּל וַיֵּשֶׁב בַּמִּדְבָּר וַיְהִי רֹבֶה קַשָּׁת. כא וַיֵּשֶׁב בְּמִדְבַּר פָּארָן וַתִּקַּח לוֹ אִמּוֹ אִשָּׁה מֵאֶרֶץ מִצְרָיִם.

בראשית כ״א — חלק ג׳: ברית אבימלך ובאר שבע (כב–לד)

כב וַיְהִי בָּעֵת הַהִוא וַיֹּאמֶר אֲבִימֶלֶךְ וּפִיכֹל שַׂר צְבָאוֹ אֶל אַבְרָהָם לֵאמֹר אֱלֹהִים עִמְּךָ בְּכֹל אֲשֶׁר אַתָּה עֹשֶׂה. כג וְעַתָּה הִשָּׁבְעָה לִּי בֵאלֹהִים הֵנָּה אִם תִּשְׁקֹר לִי וּלְנִינִי וּלְנֶכְדִּי כַּחֶסֶד אֲשֶׁר עָשִׂיתִי עִמְּךָ תַּעֲשֶׂה עִמָּדִי וְעִם הָאָרֶץ אֲשֶׁר גַּרְתָּה בָּהּ. כד וַיֹּאמֶר אַבְרָהָם אָנֹכִי אִשָּׁבֵעַ. כה וְהוֹכִחַ אַבְרָהָם אֶת אֲבִימֶלֶךְ עַל אֹדוֹת בְּאֵר הַמַּיִם אֲשֶׁר גָּזְלוּ עַבְדֵי אֲבִימֶלֶךְ. כז וַיִּקַּח אַבְרָהָם צֹאן וּבָקָר וַיִּתֵּן לַאֲבִימֶלֶךְ וַיִּכְרְתוּ שְׁנֵיהֶם בְּרִית. כח וַיַּצֵּב אַבְרָהָם אֶת שֶׁבַע כִּבְשֹׂת הַצֹּאן לְבַדְּהֶן. ל וַיֹּאמֶר כִּי אֶת שֶׁבַע כְּבָשֹׂת תִּקַּח מִיָּדִי בַּעֲבוּר תִּהְיֶה לִּי לְעֵדָה כִּי חָפַרְתִּי אֶת הַבְּאֵר הַזֹּאת. לא עַל כֵּן קָרָא לַמָּקוֹם הַהוּא בְּאֵר שָׁבַע כִּי שָׁם נִשְׁבְּעוּ שְׁנֵיהֶם. לג וַיִּטַּע אֶשֶׁל בִּבְאֵר שָׁבַע וַיִּקְרָא שָׁם בְּשֵׁם ה׳ אֵל עוֹלָם.
